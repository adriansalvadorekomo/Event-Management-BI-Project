‚ö° Sample Analytical Queries
Revenue by Season
SELECT s.season_name, SUM(f.final_price_reservation) AS revenue, AVG(f.rating_evaluation) AS avg_rating, COUNT(*) AS reservations FROM Fact_Event_Management f JOIN Dim_Season s ON f.fk_season = s.season_id JOIN Dim_Date d ON f.fk_event_date = d.date_id WHERE d.year = 2024 GROUP BY s.season_name ORDER BY revenue DESC;
Marketing ROI
SELECT s.season_name, SUM(f.marketing_spend) AS spend, SUM(f.final_price_reservation) AS revenue, ROUND( SUM(f.final_price_reservation) / NULLIF(SUM(f.marketing_spend), 0), 2) AS roi FROM Fact_Event_Management f JOIN Dim_Season s ON f.fk_season = s.season_id GROUP BY s.season_name ORDER BY roi DESC;
Competitor Benchmark
SELECT c.brand_name, AVG(f.starting_price_competitor) AS avg_price, AVG(f.rating_stars_competitor) AS avg_stars, AVG(f.rating_quality_price) AS our_qp_rating FROM Fact_Event_Management f JOIN Dim_Competitor c ON f.fk_competitor = c.competitor_id GROUP BY c.brand_name ORDER BY avg_price;
